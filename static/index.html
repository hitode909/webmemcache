WebMemcache

About

WebMemcache is web frontend of google memcache. Anyone can use from anywhere.

Usage

WebMemcache has not RESTful APIs.

Get

jQuery.ajax({
   url: 'http://webmemcache.appspot.com/get',
   type: 'GET',
   dataType: 'json',
   data: {'key': ['foo','bar'], namespace: 'test'},
   success: function(json){
     console.log(json)
   }
 });

Set

jQuery.ajax({
   url: 'http://webmemcache.appspot.com/set',
   type: 'POST',
   dataType: 'json',
   data: {'foo': 'foooo', 'bar': 'barrrrrr', namespace: 'test'},
   success: function(json){
     console.log(json)
   }
 });

Delete

jQuery.ajax({
   url: 'http://webmemcache.appspot.com/delete',
   type: 'POST',
   dataType: 'json',
   data: {'key': ['foo', 'bar'], namespace: 'test'},
   success: function(json){
     console.log(json)
   }
 });

Stats

jQuery.ajax({
   url: 'http://webmemcache.appspot.com/stats',
   type: 'GET',
   dataType: 'json',
   success: function(json){
     console.log(json)
   }
 });
